```java 反转链表
class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode next = head;			//主节点
        ListNode result = null;			//结果节点
        while(next != null) {
            ListNode temp = next.next;   //暂为主右
            next.next = result;			 //主指结
            result = next;				 //结为主
            next = temp;				 //主为暂
        }
        return result;
    }
}
```

```java 判断环链表
public class Solution {
    public boolean hasCycle(ListNode head) {
        if(head == null) return false;
        ListNode slow = head;
        ListNode fast = head;
        while(fast != null && fast.next != null) {	//注意条件
            slow = slow.next;
            fast = fast.next.next;
            if(slow == fast) return true;	//最后判断
        }
        return false;
    }
}
```

```java 两链表数相加
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode node = new ListNode();
        ListNode res = node;
        int carry = 0;
        while(l1 != null || l2 != null || carry != 0) {
            int n1 = l1 == null ? 0 : l1.val;
            int n2 = l2 == null ? 0 : l2.val;
            int temp = n1 + n2 + carry;
            carry = temp /10;
            node.next = new ListNode(temp % 10);
            node = node.next;
            if(l1 != null) l1 = l1.next;
            if(l2 != null) l2 = l2.next;
        }
        return res.next;
    }
}
```

### 双向链表

```java 纯粹的双向链表实现
public class DoublyLinkedList<T> {

    // 节点定义
    static class Node<T> {
        T value;
        Node<T> prev;
        Node<T> next;

        Node(T value) {
            this.value = value;
        }
    }

    private Node<T> head; // 虚拟头
    private Node<T> tail; // 虚拟尾
    private int size;

    public DoublyLinkedList() {
        head = new Node<>(null);
        tail = new Node<>(null);
        head.next = tail;
        tail.prev = head;
        size = 0;
    }

    // 在头部插入节点
    public void addFirst(T value) {
        Node<T> node = new Node<>(value);
        node.next = head.next;
        node.prev = head;
        head.next.prev = node;
        head.next = node;
        size++;
    }

    // 在尾部插入节点
    public void addLast(T value) {
        Node<T> node = new Node<>(value);
        node.prev = tail.prev;
        node.next = tail;
        tail.prev.next = node;
        tail.prev = node;
        size++;
    }

    // 删除指定节点
    public void remove(Node<T> node) {
        if (node == null || node == head || node == tail) return;
        node.prev.next = node.next;
        node.next.prev = node.prev;
        size--;
    }

    // 删除尾节点并返回
    public Node<T> removeLast() {
        if (tail.prev == head) return null; // 空链表
        Node<T> last = tail.prev;
        remove(last);
        return last;
    }

    // 获取链表长度
    public int getSize() {
        return size;
    }

    // 遍历打印（调试用）
    public void printList() {
        Node<T> curr = head.next;
        while (curr != tail) {
            System.out.print(curr.value + " ");
            curr = curr.next;
        }
        System.out.println();
    }
}

```

```java LRU缓存出现
class LRUCache {

    class DoubleLinkedNode{
        int key;
        int value;
        DoubleLinkedNode prev;
        DoubleLinkedNode next;

        public DoubleLinkedNode() {}

        DoubleLinkedNode(int key, int value) {
            this.key = key;
            this.value = value;
        }
    }

    DoubleLinkedNode head;
    DoubleLinkedNode tail;
    int capacity;
    int size;
    Map<Integer, DoubleLinkedNode> mymap = new HashMap<>();

    public LRUCache(int capacity) { //初始化
        this.size = 0;
        this.capacity = capacity;
        head = new DoubleLinkedNode();
        tail = new DoubleLinkedNode();
        head.next = tail;
        tail.prev = head;
    }
    
    public int get(int key) {   //查找缓存
        DoubleLinkedNode node = mymap.get(key);
        if(node == null) return -1;
        moveToHead(node);  //查找到，移动到头
        return node.value;
    }
    
    public void put(int key, int value) {   //插入
        DoubleLinkedNode node = mymap.get(key);
        if(node == null) {
            //不存在
            DoubleLinkedNode newNode = new DoubleLinkedNode(key, value);
            mymap.put(key, newNode);
            addToHead(newNode);   //新节点添加到头
            size++;
            if(size > capacity) {
                DoubleLinkedNode tailNode = removeTail();   //删除尾
                mymap.remove(tailNode.key);
                size--;
            }
        } else {
            //存在
            node.value = value;
            moveToHead(node);   //移动到头
        }
    }

    ////////////////////////

    
    private void addToHead(DoubleLinkedNode node) {
        node.prev = head;
        node.next = head.next;
        head.next.prev = node;
        head.next = node;
    }

    private void removeNode(DoubleLinkedNode node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }

    private void moveToHead(DoubleLinkedNode node) {
        removeNode(node);
        addToHead(node);
    }

    private DoubleLinkedNode removeTail() {
        DoubleLinkedNode res = tail.prev;
        removeNode(res);
        return res;
    }
}
/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache obj = new LRUCache(capacity);
 * int param_1 = obj.get(key);
 * obj.put(key,value);
 */
```

``` java 最大路径和 
class Solution {
    int max = Integer.MIN_VALUE;
    public int maxPathSum(TreeNode root) {
        if(root == null) return 0;
        dfs(root);
        return max;
    }

    public int dfs(TreeNode root) {
        if(root == null) return 0;
        int lenum = Math.max(dfs(root.left), 0);
        int rinum = Math.max(dfs(root.right), 0);
        max= Math.max(max, lenum + rinum + root.val);
        return root.val + Math.max(lenum, rinum);
    }
}
```

```java 层序遍历
class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> res = new ArrayList<>();
        if(root == null) return res;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);

        while(!q.isEmpty()) {
            int size = q.size();
            List<Integer> level = new ArrayList<>();
            for(int i = 0; i < size; i++) {
                TreeNode node = q.poll();
                level.add(node.val);
                if(node.left != null) q.offer(node.left);
                if(node.right != null) q.offer(node.right);
            }
            res.add(level);
        }
        return res;
    }
}
```

### 构造二叉树

```java 前序+中序构造二叉树
class Solution {
    private Map<Integer, Integer> inorderIndexMap; // val -> index 映射

    public TreeNode buildTree(int[] preorder, int[] inorder) {
        // 建立中序索引哈希表，方便 O(1) 查找根节点在中序的位置
        inorderIndexMap = new HashMap<>();
        for (int i = 0; i < inorder.length; i++) {
            inorderIndexMap.put(inorder[i], i);
        }
        // 从 0 到 preorder.length-1
        return helper(preorder, 0, preorder.length - 1, 0, inorder.length - 1);
    }

    // preStart, preEnd : 当前处理的先序区间
    // inStart, inEnd   : 当前处理的中序区间
    private TreeNode helper(int[] preorder, int preStart, int preEnd, int inStart, int inEnd) {
        if (preStart > preEnd || inStart > inEnd) return null;

        int rootVal = preorder[preStart];
        TreeNode root = new TreeNode(rootVal);

        int inRootIndex = inorderIndexMap.get(rootVal);
        int leftSize = inRootIndex - inStart; // 左子树节点个数

        // 构建左子树
        root.left = helper(preorder, preStart + 1, preStart + leftSize, inStart, inRootIndex - 1);
        // 构建右子树
        root.right = helper(preorder, preStart + leftSize + 1, preEnd, inRootIndex + 1, inEnd);

        return root;
    }
}
```

```java 
import java.util.ArrayDeque;
import java.util.Deque;

public class StackDemo {
    public static void main(String[] args) {
        // 创建栈（使用 ArrayDeque）
        Deque<Integer> stack = new ArrayDeque<>();

        // 1. 入栈 push
        stack.push(10);
        stack.push(20);
        stack.push(30);
        System.out.println("栈内容: " + stack); // 输出类似: [30, 20, 10]

        // 2. 查看栈顶元素 peek（不删除）
        System.out.println("栈顶元素: " + stack.peek()); // 输出: 30

        // 3. 出栈 pop（删除栈顶元素）

        System.out.println("弹出的元素: " + popped); // 输出: 30
        System.out.println("栈内容: " + stack);      // 输出类似: [20, 10]

        // 4. 判断栈是否为空 isEmpty
        System.out.println("栈是否为空: " + stack.isEmpty()); // 输出: false

        // 5. 栈大小 size
        System.out.println("栈大小: " + stack.size()); // 输出: 2

        // 6. 模拟 search（从栈顶开始的距离）
        System.out.println("20 在栈顶距离: " + searchFromTop(stack, 20));   // 输出: 1
        System.out.println("100 在栈顶距离: " + searchFromTop(stack, 100)); // 输出: -1

        // 遍历栈（Deque 默认从栈顶到栈底）
        System.out.print("栈遍历: ");
        for (int num : stack) {
            System.out.print(num + " ");
        }
        System.out.println();
    }

    // 模拟 Stack.search 的功能
    public static <T> int searchFromTop(Deque<T> stack, T target) {
        int distance = 1; // 从栈顶开始计数
        for (T item : stack) {
            if (item.equals(target)) {
                return distance;
            }
            distance++;
        }
        return -1;
    }
}
```

### 双端队列

```java 
import java.util.Deque;
import java.util.LinkedList;

public class DequeExample {
    public static void main(String[] args) {
        // 创建双端队列
        Deque<Integer> deque = new LinkedList<>();

        // --- 添加元素 ---
        deque.offerLast(1);   // 从尾部添加 1
        deque.offerLast(2);   // 从尾部添加 2
        deque.offerFirst(0);  // 从头部添加 0
        System.out.println("Deque after adding: " + deque); // [0, 1, 2]

        // --- 访问元素 ---
        System.out.println("First element: " + deque.peekFirst()); // 0
        System.out.println("Last element: " + deque.peekLast());   // 2

        // --- 移除元素 ---
        int first = deque.pollFirst(); // 从头部移除元素
        int last = deque.pollLast();   // 从尾部移除元素
        System.out.println("Removed first: " + first + ", last: " + last); // 0,2
        System.out.println("Deque after removing: " + deque); // [1]

        // --- 当作栈使用 ---
        deque.push(10);  // 等价于 offerFirst
        deque.push(20);
        System.out.println("Deque as stack: " + deque); // [20,10,1]
        int top = deque.pop(); // 弹出栈顶（队首）
        System.out.println("Popped from stack: " + top); // 20
        System.out.println("Deque now: " + deque); // [10,1]
    }
}
```
